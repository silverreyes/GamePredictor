**Create `frontend/src/components/picks/SpreadLabel.tsx`:**

```typescript
interface SpreadLabelProps {
  predictedSpread?: number;
  actualSpread?: number;
}

function formatSpread(value: number): string {
  const sign = value >= 0 ? "+" : "";
  return `${sign}${value.toFixed(1)}`;
}

function getSpreadErrorColor(error: number): string {
  if (error <= 3) return "text-green-400";
  if (error <= 7) return "text-amber-400";
  return "text-red-400";
}

export function SpreadLabel({ predictedSpread, actualSpread }: SpreadLabelProps) {
  if (predictedSpread == null) return null;

  const error =
    actualSpread != null ? Math.abs(predictedSpread - actualSpread) : null;

  return (
    <div className="flex flex-col gap-0.5">
      <p className="text-sm text-muted-foreground">
        {formatSpread(predictedSpread)} spread
      </p>
      {actualSpread != null && error != null && (
        <p className="text-xs text-muted-foreground">
          Actual {formatSpread(actualSpread)}{" "}
          <span className={`font-semibold ${getSpreadErrorColor(error)}`}>
            (off by {error.toFixed(1)})
          </span>
        </p>
      )}
    </div>
  );
}
```

Key decisions implemented:
- `predictedSpread == null` returns null (graceful degradation to v1.0 appearance per CONTEXT.md)
- Pre-game: Shows "+7.5 spread" in `text-sm text-muted-foreground` per UI-SPEC
- Post-game: Shows prediction + "Actual +10 (off by 2.5)" in `text-xs` with color-coded error per UI-SPEC
- Error thresholds: green <=3, amber 3-7, red >7 per CONTEXT.md
- Sign convention: positive = home favored (displayed as-is from backend) per Phase 8 convention
- Zero spread: "+0.0 spread" per UI-SPEC copywriting contract
